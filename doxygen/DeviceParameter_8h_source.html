22  DeviceParameter(T min, T max, unsigned int precision) :
23  max_(max), min_(min), precision_(precision) {

25  value_ = min;
26  }

34  unsigned int getPrecision() const { return precision_; }
36  const T& getValue() const { return value_; }
43  bool setValue(T value) {
44  if ( value >= min_ && value <= max_ ) {
45  // TODO implement rounding
46  value_ = round( value, precision_ );
47  return true;
48  }
49  else
50  return false;
51  }
52 
53 protected:
55  static T round(const T& value, unsigned int precision) {
56  T result;
57  // try not to perform unnescesary calculations
58  if ( precision > 0 ) {
59  double factor = std::pow(10., (int)precision);
60  T fixed = floor( value*factor + 0.5 );
61  result = fixed / factor;
62  }
63  else
64  result = floor( value + 0.5 );
65 
66  return result;
67  }
68

70  const T min_;
71  const unsigned int precision_;
72  T value_;
73 };
